#version 330 core
layout (location = 0) in vec2 pos;
layout (location = 1) in vec2 inTexCoord;
layout (location = 2) in vec2 posOffset;
layout (location = 3) in vec2 texOffset;

out vec2 texCoord;

uniform mat4 model;
uniform mat4 view;

void main() {
    gl_Position = view * model * vec4(pos + posOffset, 0.0f, 1.0f);
    texCoord = inTexCoord + texOffset;
}
